Webhooks allow you to receive real-time notifications about events that happen in your HamsukyPay account.


# Example webhook handler in Python (Flask)
@app.route('/webhooks/hamsukypay', methods=['POST'])
def hamsukypay_webhook():
    payload = request.data
    signature = request.headers.get('HamsukyPay-Signature')
    
    # Verify webhook signature
    event = hamsukypay.Webhook.construct_event(
        payload, signature, webhook_secret
    )
    
    # Handle the event
    if event.type == 'payment.succeeded':
        payment = event.data.object
        fulfill_order(payment.metadata.order_id)
    
    return '', 200
